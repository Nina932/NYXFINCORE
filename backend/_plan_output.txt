```python
# Key class and function signatures:

class ForecastEngine:
    """
    Business-specific forecasting for fuel distribution.
    Methods: Moving Average, Exponential Smoothing, Linear Regression, Growth Rate.
    Handles single-period gracefully (returns baseline projection with wide confidence).
    """

    @staticmethod
    def moving_average(
        data_points: List[float],
        periods_ahead: int = 6,
        window: int = 3,
        confidence: float = 0.95
    ) -> List[Dict[str, Any]]:
        """
        Uses pandas.rolling(window).mean() for smoothed projections.
        Returns: [{"period_index": i, "value": v, "lower": lb, "upper": ub}]
        Confidence interval: +/- z * rolling_std
        """

    @staticmethod
    def exponential_smoothing(
        data_points: List[float],
        periods_ahead: int = 6,
        alpha: float = 0.3,
        confidence: float = 0.95
    ) -> List[Dict[str, Any]]:
        """
        Simple Exponential Smoothing (manual implementation).
        S_t = alpha * Y_t + (1-alpha) * S_{t-1}
        """

    @staticmethod
    def linear_regression(
        data_points: List[float],
        periods_ahead: int = 6,
        confidence: float = 0.95
    ) -> List[Dict[str, Any]]:
        """
        Uses numpy.polyfit(x, y, deg=1) for trend line.
        Confidence interval from residual standard error.
        """

    @staticmethod
    def growth_rate(
        data_points: List[float],
        periods_ahead: int = 6,
        custom_rate: Optional[float] = None,
        confidence: float = 0.95
    ) -> List[Dict[str, Any]]:
        """
        Projects using CAGR or custom growth rate.
        CAGR = (end/start)^(1/n) - 1
        """

    @staticmethod
    def seasonal_decompose(
        data_points: List[float],
        period_labels: List[str]
    ) -> Dict[str, Any]:
        """
        Additive seasonal decomposition:
        Y = Trend + Seasonal + Residual
        Uses pandas rolling mean for trend, then computes seasonal factors.
        Returns: {"trend": [...], "seasonal": [...], "residual": [...], "seasonal_factors": {...}}
        """

    @staticmethod
    def auto_select_method(data_points: List[float]) -> str:
        """
        Select best method based on data characteristics:
        - <2 points: growth_rate (default 0%)
        - 2-3 points: linear_regression
        - 4-11 points: exponential_smoothing
        - 12+ points: moving_average (can capture seasonality)
        """

    async def generate_forecast(
        self,
        db: AsyncSession,
        forecast_type: str,        # "revenue"|"cogs"|"margin"|"expense"|"ebitda"
        product: Optional[str],
        segment: Optional[str],
        method: Optional[str],     # None = auto-select
        periods_ahead: int = 6,
        **kwargs                   # method-specific params
    ) -> Forecast:
        """
        Main entry point. Gathers historical data across multiple datasets,
        runs selected forecasting method, stores result in Forecast model.
        Returns: Forecast model instance (already persisted).

        Data gathering logic:
        1. Query all datasets ordered by period
        2. For each dataset, build IncomeStatement to get the target metric
        3. Build time series: [(period_label, value)]
        4. Handle single-period: use baseline with wide confidence
        5. Run forecast method
        6. Persist to Forecast table
        """

    async def _gather_historical_data(
        self,
        db: AsyncSession,
        forecast_type: str,
        product: Optional[str],
        segment: Optional[str]
    ) -> Tuple[List[str], List[float]]:
        """
        Gather historical values across all datasets.
        Returns (period_labels, values) sorted chronologically.
        """
```

```python
async def _generate_forecast(self, p: Dict, db: AsyncSession) -> str:
    from app.services.forecasting import ForecastEngine
    engine = ForecastEngine()
    forecast = await engine.generate_forecast(
        db=db,
        forecast_type=p.get("forecast_type", "revenue"),
        product=p.get("product"),
        segment=p.get("segment"),
        method=p.get("method"),  # None = auto
        periods_ahead=p.get("periods_ahead", 6),
    )
    # Format results as markdown + chart data
    values = forecast.values or []
    labels = [v.get("period", f"P{i+1}") for i, v in enumerate(values)]
    data = [round(v.get("value", 0), 2) for v in values]
    chart_json = json.dumps({"type": "line", "title": f"{forecast.forecast_type.title()} Forecast ({forecast.method})", "labels": labels, "data": data})

    lines = [f"**{forecast.forecast_type.title()} Forecast** (Method: {forecast.method})"]
    lines.append(f"Forecast periods: {forecast.periods}")
    if forecast.product:
        lines.append(f"Product: {forecast.product}")
    if forecast.segment:
        lines.append(f"Segment: {forecast.segment}")
    lines.append("")
    for v in values:
        ci = f" [{fgel(v.get('lower', 0))} - {fgel(v.get('upper', 0))}]" if v.get("lower") else ""
        lines.append(f"  {v.get('period', '?')}: {fgel(v.get('value', 0))}{ci}")
    lines.append(f"\n__CHART__{chart_json}__END__")
    lines.append(f"\nSaved forecast (ID: {forecast.id})")
    return "\n".join(lines)
```

```
═══ FINANCIAL REASONING FRAMEWORKS ═══
Apply these frameworks when analyzing financials:

1. DuPont Decomposition:
   ROE = (Net Profit / Revenue) * (Revenue / Total Assets) * (Total Assets / Equity)
   Use this to decompose profitability drivers when Balance Sheet data exists.

2. Margin Waterfall Analysis:
   Revenue → (-COGS) → Gross Margin → (-G&A) → EBITDA → (-D&A) → EBIT → (-Finance) → EBT → (-Tax) → Net Profit
   Always walk through each step when explaining profitability. Flag where value is lost.

3. Trend Decomposition:
   For multi-period data: Base Level + Trend + Seasonal + Residual
   Fuel demand is seasonal: heating diesel peaks in winter (Oct-Mar), gasoline peaks in summer (Jun-Sep).

4. Industry Benchmarks (Fuel Distribution - Georgia):
   - Gross Margin: 7-12% (industry avg ~9%)
   - EBITDA Margin: 3-6% (industry avg ~4.5%)
   - Retail Margin > Wholesale Margin (always, by design)
   - Inventory Turnover: 15-25x per year for fuel
   - Days Payable Outstanding: 30-60 days
   - Current Ratio: 1.0-1.5 (fuel distributors run lean)

5. Cross-Validation Rules:
   - Revenue vs COGS: each product should have both, flag mismatches
   - BS Balance: Assets = Liabilities + Equity (flag if off by >₾1)
   - Revenue Gross vs Net: Net = Gross - VAT (verify per product)
   - COGS formula: Total = Col_K + Col_L + Col_O (verify column sums)
   - Budget variance: flag anything >10% off budget

6. Ratio Analysis (compute when asked):
   - Gross Margin % = Gross Margin / Revenue * 100
   - EBITDA Margin % = EBITDA / Revenue * 100
   - Wholesale Margin % vs Retail Margin %
   - COGS as % of Revenue (by segment)
   - G&A as % of Revenue
   - Revenue per product concentration (Pareto)

═══ FORECAST INTERPRETATION ═══
When presenting forecasts:
- Always state the method used and why it was selected
- Show confidence intervals and explain their meaning
- For single-period data: explain that projections are baseline (no trend info)
- For multi-period: highlight trend direction and seasonal patterns
- Compare forecast vs budget where available
- Flag forecast anomalies (sudden jumps or drops)
```

```python
"""
RAG service for FinAI.
Primary: chromadb with sentence-transformers embeddings.
Fallback: SQLite FTS5 full-text search + keyword matching.
"""

class VectorStoreService:
    """Manages document embedding and semantic search."""

    def __init__(self):
        self._client = None
        self._collection = None
        self._use_chromadb = False
        self._init_attempted = False

    async def initialize(self):
        """Try to initialize chromadb; fall back to FTS5."""
        try:
            import chromadb
            self._client = chromadb.PersistentClient(path="./chromadb_data")
            self._collection = self._client.get_or_create_collection(
                name="finai_documents",
                metadata={"hnsw:space": "cosine"}
            )
            self._use_chromadb = True
        except ImportError:
            logger.info("chromadb not installed; using FTS5 fallback")
            self._use_chromadb = False
        self._init_attempted = True

    async def index_documents(self, documents: List[Dict], db: AsyncSession):
        """
        Index a batch of documents.
        Each doc: {"id": str, "content": str, "metadata": dict, "document_type": str}
        If chromadb available: embed and upsert.
        If FTS5 fallback: store in FinancialDocument table (FTS via LIKE queries).
        """

    async def index_dataset(self, dataset_id: int, db: AsyncSession):
        """
        Index all data from a dataset:
```

```python
class AnomalyDetector:
    """Statistical anomaly detection using numpy + pandas only."""

    def zscore_detection(
        self,
        transactions: List[Dict],
        threshold: float = 2.0,
        group_by: str = "cost_class"
    ) -> List[Dict]:
        """
        Z-score: flag transactions > threshold standard deviations from category mean.
        Groups by cost_class, dept, or counterparty.
        Returns: [{"transaction_id": ..., "score": z, "severity": ..., "description": ...}]
        """
        # Use numpy: mean = np.mean(amounts), std = np.std(amounts)
        # z = (x - mean) / std
        # severity: >3sigma = "critical", >2.5 = "high", >2 = "medium"

    def iqr_detection(
        self,
        transactions: List[Dict],
        multiplier: float = 1.5
    ) -> List[Dict]:
        """
        IQR method: flag transactions below Q1-1.5*IQR or above Q3+1.5*IQR.
        Uses numpy: Q1 = np.percentile(amounts, 25), Q3 = np.percentile(amounts, 75)
        """

    def benford_law(
        self,
        transactions: List[Dict]
    ) -> Dict:
        """
        Benford's Law: compare first-digit distribution to expected.
        Expected: P(d) = log10(1 + 1/d) for d=1..9
        Uses chi-squared test: chi2 = sum((observed - expected)^2 / expected)
        Returns: {"digits": {1: {"expected": 0.301, "observed": 0.285, "deviation": -0.016}}, "chi_squared": ..., "suspicious": bool, "flagged_digits": [...]}
        """

    def seasonal_anomaly(
        self,
        historical_data: List[Dict],  # [{period, value}]
        current_value: float,
        current_period: str,
        threshold: float = 2.0
    ) -> Dict:
        """
        Compare current value against seasonal expectation.
        Uses historical data to compute expected seasonal value.
        """

        self,
```

## Risk Areas and Mitigation

1. **Single-period data**: The forecasting engine must handle gracefully. Mitigation: return baseline projection with wide confidence intervals and a warning message.

2. **chromadb dependency size**: chromadb pulls in sentence-transformers (large). Mitigation: graceful fallback to FTS5 keyword search if not installed. Make it optional in requirements.

3. **Performance of lineage tracking**: Adding a DataLineage row for every parsed record roughly doubles DB writes on upload. Mitigation: use `db.add_all()` for bulk inserts, consider making lineage optional via config flag.

4. **Scheduler reliability**: `asyncio.create_task()` runs in the same process. If the process crashes, scheduled reports are lost. Mitigation: store `next_run` in DB so scheduler picks up missed reports on restart.

5. **Exchange rate API limits**: Free tier may have rate limits. Mitigation: cache aggressively (one fetch per day per currency pair), store in ExchangeRate table.

6. **Frontend file size**: The HTML file is already 392KB. Adding 5 new page sections will increase it further. Mitigation: keep new UI sections minimal, use lazy rendering (only build DOM when page is navigated to).
